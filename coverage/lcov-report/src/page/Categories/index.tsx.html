export const getTitleCategorie = (typeWine: string | undefined): string => {
  switch (typeWine) {
    case "Tinto":
      return "Un poco de historia del Vino Tinto";
 
    case "Rosado":
      return "Un poco de historia del Vino Rosado";
 
    case "Blanco":
      return "Un poco de historia del Vino Blanco";
 
    default:
      return "Un poco de historia del Vino Tinto";
  }
};
